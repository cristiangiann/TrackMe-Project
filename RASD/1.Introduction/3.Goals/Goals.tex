\section{Goals}

\paragraph{Data4Help}
\begin{itemize}
	
	\item[G\subs{1}]Collect \ic{User data} through \ic{Smart Wearables}.
	
	\item[G\subs{2}]Send specific \ic{User data} to \ic{Third parties} only if \ic{User} consent was given after\ic{Third party} access request.
	\item[G\subs{3}]Send anonimyzed requested \ic{Group Data} to \ic{Third parties} if the group it refers to is made up of 1000 or more \ic{Users}.

	\item[G\subs{4}]Send \ic{User data} and \ic{Group Data} to subscribed authorized \ic{Third parties} as soon as they are produced.
	
	\item[G\subs{5}]Allow \ic{Users} to manage their subscription to \ic{Services} and to Data4Help.		
	
\end{itemize}

\paragraph{AutomatedSOS}
\begin{itemize}
	\item[G\subs{6}]Analyze \ic{User data} to check whether or not a \ic{User} is a \ic{User in need}.
	\item[G\subs{7}]Send an ambulance to the last position of a \ic{User in need}.
\end{itemize}

\paragraph{Track4Run}
\begin{itemize}
	\item[G\subs{8}]Allow \ic{Organizers} to create a \ic{Run}.
	\item[G\subs{9}]Allow \ic{Users} to enroll in a \ic{Run} as \ic{Participants}. 
	\item[G\subs{10}]Allow \ic{Spectators} to watch a \ic{Run}.
\end{itemize}
